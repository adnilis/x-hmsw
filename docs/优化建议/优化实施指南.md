```go
package simd

import "golang.org/x/sys/cpu"

// CPUInfo CPU特性信息
var CPUInfo = struct {
    HasAVX2  bool
    HasAVX   bool
    HasSSE2  bool
    HasSSE41 bool
}{
    HasAVX2:  cpu.X86.HasAVX2,
    HasAVX:   cpu.X86.HasAVX,
    HasSSE2:  cpu.X86.HasSSE2,
    HasSSE41: cpu.X86.HasSSE41,
}

// OptimalDotProduct 选择最优的点积实现
func OptimalDotProduct(a, b []float32) float32 {
    if CPUInfo.HasAVX2 {
        return DotProductAVX2(a, b)
    }
    if CPUInfo.HasSSE2 {
        return DotProductSSE2(a, b)
    }
    return DotProduct(a, b) // 通用实现
}
```

## 阶段2: Heap优化 (优先级: 高)

### 步骤2.1: 实现优化的Heap

**文件**: `indexes/hnsw/optimized_heap.go` (新建)

```go
package hnsw

import (
    "math"
)

// FastMaxHeap 快速最大堆实现
type FastMaxHeap struct {
    data []*candidate
}

func NewFastMaxHeap(capacity int) *FastMaxHeap {
    return &FastMaxHeap{
        data: make([]*candidate, 0, capacity),
    }
}

func (h *FastMaxHeap) Len() int {
    return len(h.data)
}

func (h *FastMaxHeap) Less(i, j int) bool {
    return h.data[i].Distance > h.data[j].Distance
}

func (h *FastMaxHeap) Swap(i, j int) {
    h.data[i], h.data[j] = h.data[j], h.data[i]
}

func (h *FastMaxHeap) Push(x interface{}) {
    h.data = append(h.data, x.(*candidate))
    h.up(len(h.data) - 1)
}

func (h *FastMaxHeap) Pop() interface{} {
    n := len(h.data)
    item := h.data[0]
    h.data[0] = h.data[n-1]
    h.data = h.data[:n-1]
    if len(h.data) > 0 {
        h.down(0)
    }
    return item
}

func (h *FastMaxHeap) Peek() interface{} {
    if len(h.data) == 0 {
        return nil
    }
    return h.data[0]
}

// up 向上调整
func (h *FastMaxHeap) up(j int) {
    for {
        i := (j - 1) / 2 // parent
        if i == j || !h.Less(j, i) {
            break
        }
        h.Swap(i, j)
        j = i
    }
}

// down 向下调整
func (h *FastMaxHeap) down(i int) {
    n := h.Len()
    for {
        j1 := 2*i + 1
        if j1 >= n || j1 < 0 {
            break
        }
        j := j1
        if j2 := j1 + 1; j2 < n && h.Less(j2, j1) {
            j = j2
        }
        if !h.Less(j, i) {
            break
        }
        h.Swap(i, j)
        i = j
    }
}
```

```go
func (h *HNSWGraph) Insert(id int, vector []float32) *Node {
    // 计算并缓存归一化向量
    normalizedVector := make([]float32, len(vector))
    copy(normalizedVector, vector)

    // 归一化
    var norm float32
    for _, v := range normalizedVector {
        norm += v * v
    }
    norm = float32(math.Sqrt(float64(norm)))
    if norm > 0 {
        for i := range normalizedVector {
            normalizedVector[i] /= norm
        }
```

### 步骤4.1: 实现细粒度锁

**文件**: `indexes/hnsw/concurrent_opt.go` (新建)

```go
package hnsw

import (
    "sync"
    "unsafe"
)

// ConcurrentHNSW 并发优化的HNSW
type ConcurrentHNSW struct {
    *HNSWGraph
    // 每层一个锁
    levelLocks []sync.RWMutex
    // 节点锁池
    nodeLockPool sync.Pool
}

func NewConcurrentHNSW(dim, M, efConstruction, maxNodes int, distanceFunc func(a, b []float32) float32) *ConcurrentHNSW {
    graph := NewHNSW(dim, M, efConstruction, maxNodes, distanceFunc)

    return &ConcurrentHNSW{
        HNSWGraph:   graph,
        levelLocks:  make([]sync.RWMutex, 32),
        nodeLockPool: sync.Pool{
            New: func() interface{} {
                return &sync.RWMutex{}
            },
        },
    }
}

// 获取节点锁
func (h *ConcurrentHNSW) getLock(nodeID int) *sync.RWMutex {
    return h.levelLocks[nodeID%len(h.levelLocks)]
}

// 并发搜索
func (h *ConcurrentHNSW) ConcurrentSearch(query []float32, k int) []SearchResult {
    normalizedQuery := h.normalizeVector(query)

    h.mu.RLock()
    entryPoint := h.EntryPoint
    currentMaxLevel := h.MaxLevel
    h.mu.RUnlock()

    // 并发搜索各层
    resultsChan := make(chan []SearchResult, currentMaxLevel)

    wg := sync.WaitGroup{}
    for l := currentMaxLevel; l >= 0; l-- {
        wg.Add(1)
        go func(level int, ep *Node) {
            defer wg.Done()
            results := h.searchLayer(normalizedQuery, ep, h.EfSearch, level)
            resultsChan <- results
        }(l, entryPoint)
    }

    go func() {
        wg.Wait()
        close(resultsChan)
    }()

    // 合并结果
    var allResults []SearchResult
    for results := range resultsChan {
        allResults = append(allResults, results...)
    }

}
```
